### 表关系说明

| 关系 | 父表 | 子表 | 类型 | 说明 |
|------|------|------|------|------|
| R1 | users | devices | 1:N | 一个用户可绑定多台设备 |
| R2 | users | sleep_reports | 1:N | 一个用户有多条睡眠报告 |
| R3 | users | user_settings | 1:1 | 一个用户对应一条设置记录 |
| R4 | users | doctor_authorizations | 1:N | 患者端：一个用户可授权多个医生 |
| R5 | users | doctor_authorizations | 1:N | 医生端：一个医生可被多个患者授权 |

## 二、DDL（数据定义语句）

### 2.1 用户表 — users

```sql
CREATE TABLE `users` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '用户ID',
  `phone` VARCHAR(20) NOT NULL COMMENT '手机号',
  `password_hash` VARCHAR(255) NOT NULL COMMENT '密码哈希',
  `nickname` VARCHAR(50) DEFAULT '' COMMENT '昵称',
  `avatar_url` VARCHAR(500) DEFAULT '' COMMENT '头像URL',
  `gender` TINYINT DEFAULT 0 COMMENT '性别：0=未知 1=男 2=女',
  `birth_date` DATE DEFAULT NULL COMMENT '出生日期',
  `height` DECIMAL(5,1) DEFAULT NULL COMMENT '身高(cm)',
  `weight` DECIMAL(5,1) DEFAULT NULL COMMENT '体重(kg)',
  `role` TINYINT NOT NULL DEFAULT 0 COMMENT '角色：0=普通用户 1=医生 2=管理员',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '软删除：0=正常 1=已删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_users_phone` (`phone`),
  KEY `idx_users_role` (`role`),
  KEY `idx_users_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户表';
```

### 2.2 设备表 — devices

```sql
CREATE TABLE `devices` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '设备ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '绑定用户ID',
  `device_name` VARCHAR(100) NOT NULL COMMENT '设备名称',
  `device_type` VARCHAR(50) NOT NULL COMMENT '设备类型：watch/band/ring/mat',
  `device_mac` VARCHAR(50) DEFAULT '' COMMENT '设备MAC地址',
  `firmware_version` VARCHAR(20) DEFAULT '' COMMENT '固件版本',
  `is_online` TINYINT NOT NULL DEFAULT 0 COMMENT '在线状态：0=离线 1=在线',
  `last_sync_at` DATETIME DEFAULT NULL COMMENT '最后同步时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '软删除：0=正常 1=已删除',
  PRIMARY KEY (`id`),
  KEY `idx_devices_user_id` (`user_id`),
  KEY `idx_devices_device_type` (`device_type`),
  KEY `idx_devices_is_online` (`is_online`),
  KEY `idx_devices_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='设备表';
```

### 2.3 睡眠报告表 — sleep_reports

```sql
CREATE TABLE `sleep_reports` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '报告ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `report_date` DATE NOT NULL COMMENT '报告日期',
  `fall_asleep_time` TIME DEFAULT NULL COMMENT '入睡时间',
  `wake_up_time` TIME DEFAULT NULL COMMENT '起床时间',
  `total_sleep_minutes` INT UNSIGNED DEFAULT 0 COMMENT '总睡眠时长(分钟)',
  `deep_sleep_minutes` INT UNSIGNED DEFAULT 0 COMMENT '深睡时长(分钟)',
  `light_sleep_minutes` INT UNSIGNED DEFAULT 0 COMMENT '浅睡时长(分钟)',
  `rem_sleep_minutes` INT UNSIGNED DEFAULT 0 COMMENT 'REM睡眠时长(分钟)',
  `awake_count` INT UNSIGNED DEFAULT 0 COMMENT '夜间醒来次数',
  `heart_rate_avg` DECIMAL(4,1) DEFAULT NULL COMMENT '平均心率(bpm)',
  `blood_oxygen_avg` DECIMAL(4,1) DEFAULT NULL COMMENT '平均血氧(%)',
  `sleep_score` TINYINT UNSIGNED DEFAULT NULL COMMENT '睡眠评分(0-100)',
  `suggestion` TEXT COMMENT '睡眠建议',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '软删除：0=正常 1=已删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_sleep_reports_user_date` (`user_id`, `report_date`),
  KEY `idx_sleep_reports_report_date` (`report_date`),
  KEY `idx_sleep_reports_sleep_score` (`sleep_score`),
  KEY `idx_sleep_reports_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='睡眠报告表';
```

### 2.4 用户设置表 — user_settings

```sql
CREATE TABLE `user_settings` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '设置ID',
  `user_id` BIGINT UNSIGNED NOT NULL COMMENT '用户ID',
  `target_bedtime` TIME DEFAULT NULL COMMENT '目标入睡时间',
  `target_wake_time` TIME DEFAULT NULL COMMENT '目标起床时间',
  `target_sleep_hours` DECIMAL(3,1) DEFAULT NULL COMMENT '目标睡眠时长(小时)',
  `remind_bedtime` TINYINT NOT NULL DEFAULT 1 COMMENT '入睡提醒开关：0=关 1=开',
  `remind_wakeup` TINYINT NOT NULL DEFAULT 1 COMMENT '起床提醒开关：0=关 1=开',
  `bedtime_remind_min` INT UNSIGNED DEFAULT 30 COMMENT '入睡提醒提前分钟数',
  `data_share_scope` TINYINT NOT NULL DEFAULT 0 COMMENT '数据分享范围：0=仅自己 1=医生可见 2=公开',
  `push_enabled` TINYINT NOT NULL DEFAULT 1 COMMENT '推送通知：0=关 1=开',
  `do_not_disturb_start` TIME DEFAULT NULL COMMENT '免打扰开始时间',
  `do_not_disturb_end` TIME DEFAULT NULL COMMENT '免打扰结束时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '软删除：0=正常 1=已删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_settings_user_id` (`user_id`),
  KEY `idx_user_settings_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户设置表';
```

### 2.5 医生授权表 — doctor_authorizations

```sql
CREATE TABLE `doctor_authorizations` (
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '授权ID',
  `patient_id` BIGINT UNSIGNED NOT NULL COMMENT '患者用户ID',
  `doctor_id` BIGINT UNSIGNED NOT NULL COMMENT '医生用户ID',
  `status` TINYINT NOT NULL DEFAULT 1 COMMENT '授权状态：0=已取消 1=有效 2=已过期',
  `authorized_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '授权时间',
  `expired_at` DATETIME DEFAULT NULL COMMENT '授权过期时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `updated_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  `is_deleted` TINYINT NOT NULL DEFAULT 0 COMMENT '软删除：0=正常 1=已删除',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_doctor_auth_patient_doctor` (`patient_id`, `doctor_id`),
  KEY `idx_doctor_auth_doctor_id` (`doctor_id`),
  KEY `idx_doctor_auth_status` (`status`),
  KEY `idx_doctor_auth_is_deleted` (`is_deleted`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='医生授权表';
```

## 三、索引设计说明

| 表名 | 索引名 | 类型 | 字段 | 用途 |
|------|--------|------|------|------|
| users | uk_users_phone | UNIQUE | phone | 手机号唯一登录 |
| users | idx_users_role | NORMAL | role | 按角色筛选（医生/用户） |
| devices | idx_devices_user_id | NORMAL | user_id | 查询用户的设备列表 |
| devices | idx_devices_device_type | NORMAL | device_type | 按设备类型筛选 |
| sleep_reports | uk_sleep_reports_user_date | UNIQUE | user_id, report_date | 每人每天仅一条报告 |
| sleep_reports | idx_sleep_reports_sleep_score | NORMAL | sleep_score | 按评分排序/筛选 |
| user_settings | uk_user_settings_user_id | UNIQUE | user_id | 每人仅一条设置 |
| doctor_authorizations | uk_doctor_auth_patient_doctor | UNIQUE | patient_id, doctor_id | 避免重复授权 |
| doctor_authorizations | idx_doctor_auth_doctor_id | NORMAL | doctor_id | 查询医生的患者列表 |
